44 class BasicWarehouse extends Warehouse{
45   private var products: Map[Product,Int] = Map[Product,Int]()
46   override def supply(p: Product, qty: Int): Unit = products += p -> (products.getOrElse(p,0) + qty)
47   override def get(p: Product, qty: Int): (Product,Int) = {
48     val availability = products.getOrElse(p, 0)
49     availability match {
50       case 0 => (p, 0)
51       case _ if availability >= qty => {
52         val newAvailability = availability - qty
53         products += p -> newAvailability
54         (p, qty)
55       }
56       case _ if qty > availability => {
57         products += p -> 0
58         (p, availability)
59       }
60     }
61   }

64 class Shopping(private val warehouse: Warehouse,
65                private val catalog: Catalog,
66                private var cart: Cart,
67                private val logger: Logger) {
68   def pick(p: Product, qty: Int): Cart = {
69     assert(qty>0) // precondition
70 
71     logger.log(s"Trying to pick $qty pieces of $p.")
72     val (_, howMany) = warehouse.get(p, qty)
73     if(howMany>0) {
74       logger.log(s"The warehouse has $howMany pieces; adding them to cart.")
75       val price = catalog.priceFor(p, howMany)
76       logger.log(s"$howMany pieces of $p collectively cost $price.")
77       val item = Item(p, ItemDetails(howMany, price))
78       cart.add(item)
79       logger.log(s"Updated cart: now it contains ${cart.size} items for total ${cart.totalCost}")
80     } else{
81       logger.log("There are no pieces of the requested product in the warehouse.")
82     }
83     cart
84   }
